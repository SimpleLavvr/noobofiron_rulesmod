add_namespace = noi_usa


# it's warring time
country_event = {
   id = noi_usa.1
   title = noi_usa.1.t
   desc = noi_usa.1.d
   picture = GFX_report_event_usa_heavy_cruiser

   trigger = {
        tag = USA
        has_idea = USA_enforced_neutrality
        OR = {
            has_war = yes
            date > 1941.6.1
        }
   }

   fire_only_once = yes

   option = {
        name = noi_usa.1.a
        remove_ideas = USA_enforced_neutrality
   }
}

# Sealion's happening, US joins the allies
country_event = {
    id = noi_usa.4
    title = noi_usa.4.t
    desc = noi_usa.4.d
    picture = GFX_report_event_generic_handshake

    trigger = {
        tag = USA
        any_state = {
            is_owned_by = ENG
            is_in_home_area = yes
            NOT = { is_fully_controlled_by = ENG }
        }
    }

    fire_only_once = yes

    option = {
        name = noi_usa.4.a

        IF = {
            limit = {
                has_idea = USA_enforced_neutrality
            }
            remove_ideas = USA_enforced_neutrality
        }
        ENG = {add_to_faction = USA}
    }
}

add_namespace = noi_sov

# Gives USSR military access to Spain if it has volunteers, because PDX can't code
country_event = {
    id = noi_sov.1
    title = noi_sov.1.t
    desc = noi_sov.1.d
    picture = GFX_report_event_generic_handshake
    fire_only_once = yes

    trigger = {
        tag = SOV
        any_country = {
            original_tag = SPR
            has_volunteers_amount_from = {
                tag = SOV
                count > 0
            }
        }
    }

    option = {
        name = noi_sov.1.a

        every_country = {
            limit = {
                original_tag = SPR
                has_volunteers_amount_from = {
                    tag = SOV
                    count > 0
                }
            }
            give_military_access = SOV
        }
    }
}
